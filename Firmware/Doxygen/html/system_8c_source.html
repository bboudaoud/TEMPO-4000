249 void sysSample(void)
250 {
251 
252  axisData d = {0,0,0};
253  fsRetCode retval;
254  int i = 0;
255 
256  if(!sessionInProgress) return; // Assure a session is in progress
257 
258  // Check/record any accelerometer channels
259  if(samp.axis.x_acc | samp.axis.y_acc | samp.axis.z_acc){
260  d = mpuGetAccel();
261  retval = fsWriteData((unsigned char *)&d, sizeof(axisData));
262  }
263  if(retval == FS_FAIL_BF){
264  sensBuffFull = 1;
265  return;
266  }
267 
268  // Check/record any gyro channels
269  if(samp.axis.x_gyro | samp.axis.y_gyro | samp.axis.z_gyro){
270  d = mpuGetGyro();
271  retval = fsWriteData((unsigned char *)&d, sizeof(axisData));
272  }
273  if(retval == FS_FAIL_BF){
274  sensBuffFull = 1;
275  return;
276  }
277 
278  // Check/record the temperature channel
279  if(samp.axis.temp){
280  i = mpuGetTemp();
281  retval = fsWriteData((unsigned char *)&i, sizeof(int));
282  }
283  if(retval == FS_FAIL_BF){
284  sensBuffFull = 1;
285  return;
286  }
287  // Check for a card full failure
288  if(retval == FS_FAIL_CF){
289  cardFull = 1; // Set the cardfull indicator flag
290  evtQuePut(SYS_END_SESSION); // Add an end-session to the event queue
291  return;
292  }
293  else {
294  sampleCount++; // Increment sample counter
295  sampleCount %= BLINK_COUNT; // Wrap sample counter
296  if(sampleCount== 0){ // Check for counter wrap around
297  LED_GREEN_TOGGLE(); // Toggle the LED to indicate sampling
298  }
299  }
300  return;
301 }

449 void updateState(void)
450 {
451  extern unsigned char processingCmd; // Processing command flag from command.c
452  extern cardInfo cardData; // Card data field from filesystem.c
453 
454  ftdiPacket inPacket; // In packet read from FTDI (USCI A0 UART Mode)
455 
456  unsigned int i = 0;
457 
458  switch(state){
459  case STATE_COLLECTING: // COLLECTION MODE: Check for entering charge, card full, or data overflow
460  if(CHARGING){
461  closeCause = sess_closed_chg;
462  evtQuePut(SYS_END_SESSION);
463  }
464  else if(cardFull){
465  evtQueInit();
466  closeCause = sess_closed_full;
467  evtQuePut(SYS_END_SESSION);
468  }
469  else if(sensBuffFull){
470  closeCause = sess_closed_ovfl;
471  evtQuePut(SYS_END_SESSION);
472  }
473  break;
474  case STATE_CMD: // COMMAND MODE: Check for leaving charger or receiving info
475  if(!CHARGING){ // Check for left charger
476  evtQueInit(); // Clear the event queue
477  ftdiFlush(); // Flush the FTDI comm buffer
478  if(dataCollectionEn){
479  evtQuePut(SYS_START_SESSION);
480  }
481  else{
482  evtQuePut(SYS_ENTER_SLEEP);
483  }
484  }
485  // Command processing
486  else if(!processingCmd && ftdiGetBuffSize() >= CMD_LEN){
487  for(i = 0; i < FTDI_OPEN_RETRIES; i++){ // Wait for module to become available
488  if(ftdiGetStatus() == OPEN) break;
489  }
490  inPacket = ftdiRead(CMD_LEN); // Read in a command from the buffer
491  for(i = 0; i < CMD_LEN; i++){
492  if(inPacket.data[i] == CMD_START_OF_SEQ) break;
493  }
494  if(i != CMD_LEN){
495  cmd = (cmdPkt *)(&inPacket.data[i]); // Cast in packet
496  cmd->command = (cmd->command)>>8; // Shift over command field
497  LED_GREEN_ON(); // Turn LED on
498  runCommand(cmd); // Process command and send response
499  LED_GREEN_OFF(); // Turn LED off
500  }
501  }
502  break;
503  case STATE_IDLE:
504  if(CHARGING){
505  evtQuePut(SYS_ENTER_CMD);
506  }
507  break;
508  case STATE_SLEEP:
509  break;
510  default:
511  break;
512 
513  }
