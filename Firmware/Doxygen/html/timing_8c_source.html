48 long setFLL(unsigned long TargetFreq)
49 {
50  unsigned int fllMult = (unsigned int)(TargetFreq >> 15); // fllMult = TargetDCO/32768
51 
52  // Basic Universal Clock System (UCS) Init
53  UCSCTL3 = SELREF_2; // Set FLL Reference to REF0 (internal reference oscillator)
54  UCSCTL4 = SELA_2 + SELS_4 + SELM_4; // Set ACLK = REF0, SMCLK = MCLK = DCOCLKDIV
55 
56  if(fllMult > 1024) // If FLL multiplier cannot be represented in 10 bits
57  return -1; // Return failure
58 
59  __bis_SR_register(SCG0); // Disable the FLL control loop
60  UCSCTL0 = 0; // Set lowest possible DCOx and MODx bits
61 
62  // DCO Resistor Selection (RSEL)
63  if(fllMult <= 30) // TargetFreq < 1MHz
64  UCSCTL1 = DCORSEL_0;
65  else if(fllMult <= 62) // 1MHz < TargetFreq < 2MHz
66  UCSCTL1 = DCORSEL_1;
67  else if(fllMult <= 123) // 2MHz < TargetFreq < 4MHz
68  UCSCTL1 = DCORSEL_2;
69  else if(fllMult <= 245) // 4MHz < TargetFreq < 8 MHz
70  UCSCTL1 = DCORSEL_3;
71  else if(fllMult <= 490) // 8MHz < TargetFreq < 16MHz
72  UCSCTL1 = DCORSEL_4;
73  else if(fllMult <= 611) // 16MHz < TargetFreq < 20MHz
74  UCSCTL1 = DCORSEL_5;
75  else // 20MHz < TargetFreq < 33MHz
76  UCSCTL1 = DCORSEL_6;
77 
78  UCSCTL2 = fllMult & FLLN_MASK; // Set FLLN (FLL Multiplier)
79 
80  __bic_SR_register(SCG0); // Enable the FLL control loop
81  __delay_cycles(CLOCK_STAB_PERIOD); // Delay and let clock stabilize
82 
83  // Loop until XT1,XT2 & DCO fault flag is cleared
84  do {
85  UCSCTL7 &= ~(XT2OFFG + XT1LFOFFG + DCOFFG); // Clear XT2,XT1,DCO fault flags
86  SFRIFG1 &= ~OFIFG; // Clear fault flags
87  } while (SFRIFG1&OFIFG); // Test oscillator fault flag
88 
89  return (fllMult << 15); // Return DCO frequency
90 }
