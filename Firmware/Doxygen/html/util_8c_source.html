50 {
51

57  sum1 = 0xff;
58  sum2 = 0xff;
59  } else { // Parse the previous two component sums out of the old checksum
60  sum2 = checksum >> 8;
61  sum1 = checksum & 0xff;
62  }
63

65

67  int tlen = len > 20 ? 20 : len; // Require tlen < 20 to avoid second order accumulation overflow
68  len -= tlen;

70  sum1 += *data++;
71  sum2 += sum1;
72  } while (--tlen);
73 
74  // The reduction step below is equivalent to a partial modulo 255 (may have a 1 in the upper byte)
75  // This works by taking sum1 % 256 then adding a 1 for each 256 in sum1 (sum1/256 = sum1 >> 8)
76

79  }

82
